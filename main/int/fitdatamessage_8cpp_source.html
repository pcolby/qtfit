144 /*!
145  * \internal
146  *
147  * Set the current FIT data message's fields.
148  *
149  * The method iterates through the \a record, invoking the virtual setField method to assign the
150  * extraced values to the relevant data message fields. Derived classes are expected to override
151  * setField to implement that field number to class member mapping.
152  *
153  * \param defn Data field definition describing the \a record layout.
154  * \param record The FIT data record to read fields from.
155  *
156  * \return \c true if all fields were parsed safely.
157  *
158  * \sa setField
159  */
160 bool FitDataMessagePrivate::setFields(const DataDefinition * const defn, const QByteArray &record)
161 {
162  Q_ASSERT(defn->globalMessageNumber == this->globalMessageNumber);
163  int dataOffset=0; // Next field's offset within dataRecord.
164  for (const FieldDefinition &field: defn->fieldDefinitions) {
165  if (!setField(field.number, record.mid(dataOffset,field.size), field.baseType,
166  (defn->architecture == Architecture::BigEndian)))
167  return false;
168  dataOffset += field.size;
169  }
170  return true;
171 }

173 /*!
174  * \internal
175  *
176  * Set a value of a data message field.
177  *
178  * This base implementation simply logs a warning, and ignores the field. Derived classes should
179  * override this implementation to set the relevant class members.
180  *
181  * \param fieldId The field number within the given FIT data message.
182  * \param data The raw data to extract the field value from.
183  * \param baseType The FIT base type for the field.
184  * \param bigEndian Whether or not multibyte values in \a record are big-endian.
185  *
186  * \return \c true if the field was set, or skipped; \c false if the field was corrupt or invalid.
187  */
188 bool FitDataMessagePrivate::setField(const int fieldId, const QByteArray &data,
189  const FitBaseType baseType, const bool bigEndian)
190 {
191  Q_UNUSED(bigEndian)
192  qWarning() << "ignoring unknown field" << fieldId << data << baseType;
193  return true; // True because we "handled it safely", and can continue parsing.
194 }
