79 bool FitDataMessagePrivate::setFields(const DataDefinition * const defn, const QByteArray &record)
80 {
81  Q_ASSERT(defn->globalMessageNumber == this->globalMessageNumber);
82  int dataOffset=0; // Next field's offset within dataRecord.
83  for (const FieldDefinition &field: defn->fieldDefinitions) {
84  if (!setField(field.number, record.mid(dataOffset,field.size), field.baseType,
85  (defn->architecture == Architecture::BigEndian)))
86  return false;
87  dataOffset += field.size;
88  }
89  return true;
90 }
