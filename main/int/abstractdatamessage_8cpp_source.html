99 /*!
100  * \internal
101  *
102  * Sets the current FIT data message's fields.
103  *
104  * The method iterates through the \a record, invoking the virtual setField method to assign the
105  * extraced values to the relevant data message fields. Derived classes are expected to override
106  * setField to implement that field number to class member mapping.
107  *
108  * \param defn Data field definition describing the \a record layout.
109  * \param record The FIT data record to read fields from.
110  *
111  * \return \c true if all fields were parsed safely.
112  *
113  * \sa setField
114  */
115 bool AbstractDataMessagePrivate::setFields(const DataDefinition * const defn, const QByteArray &record)
116 {
117  Q_ASSERT(defn->globalMessageNumber == this->globalMessageNumber);
118  int dataOffset=0; // Next field's offset within dataRecord.
119  for (const FieldDefinition &field: defn->fieldDefinitions) {
120  if (!setField(field.number, record.mid(dataOffset,field.size), field.baseType,
121  (defn->architecture == Architecture::BigEndian)))
122  return false;
123  dataOffset += field.size;
124  }
125  return true;
126 }

145 /*!
146  * \internal
147  *
148  * Verifies that \a actual matches \a expected, and reports a warning if not.
149  *
150  * \param actual Actual base type.
151  * \param expected Expected base type.
152  * \param name Name of the field being verfied.
153  *
154  * \return \c true if \a actual and \a expected match, \c false otherwise.
155  */
156 inline bool verifyBaseType(const FitBaseType actual, const FitBaseType expected, const char *name)
157 {
158  if (actual == expected) return true;
159  qWarning() << name << "has base type" << actual << "but should be" << expected;
160  return false;
161 }

163 /*!
164  * \internal
165  *

167  *

169  * \param expectedSize Exepcted size of \a data.

174 inline bool verifyDataSize(const QByteArray &data, const int expectedSize, const char *name)

176  if (data.size() == expectedSize) return true;
177  qWarning() << name << "size is" << data.size() << "but should be" << expectedSize;
178  return false;
