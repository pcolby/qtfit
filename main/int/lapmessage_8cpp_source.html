1443  const int fieldId, const QByteArray &data, const FitBaseType baseType, const bool bigEndian)
1444 {

1448  this->messageIndex = static_cast<MessageIndex>(bigEndian ? qFromBigEndian<quint16>(data) : qFromLittleEndian<quint16>(data));
1449  break;
1450  case 253: // See Profile.xlsx::Messages:lap.timestamp
1451  if (!verify(data, baseType, 4, FitBaseType::Uint32, "lap.timestamp")) return false;
1452  this->timestamp = static_cast<DateTime>(bigEndian ? qFromBigEndian<quint32>(data) : qFromLittleEndian<quint32>(data));
1453  break;
1454  case 0: // See Profile.xlsx::Messages:lap.event
1455  if (!verify(data, baseType, 1, FitBaseType::Enum, "lap.event")) return false;
1456  this->event = static_cast<Event>(data.at(0));
1457  break;
1458  case 1: // See Profile.xlsx::Messages:lap.eventType
1459  if (!verify(data, baseType, 1, FitBaseType::Enum, "lap.eventType")) return false;
1460  this->eventType = static_cast<EventType>(data.at(0));
1461  break;
