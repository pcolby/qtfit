188 /*!
189  * \internal
190  *
191  * Calculates a checksum, as per the algorithm used by FIT file headers.
192  *
193  * \todo Move this somewhere appropriate (probably FitStreamReaderPrivate).
194  *
195  * \param data FIT file header to calculate the checksum for.
196  *
197  * \return 16-bit checksum for \a data.
198  */
199 quint16 fitChecksum(const QByteArray &data) {
200  quint16 checksum=0;
201  for (const auto &byte: data) {
202  static const quint16 crcTable[16] = {
203  0x0000, 0xCC01, 0xD801, 0x1400, 0xF001, 0x3C00, 0x2800, 0xE401,
204  0xA001, 0x6C00, 0x7800, 0xB401, 0x5000, 0x9C01, 0x8801, 0x4400
205  };
206  for (int byteShift=0; byteShift<=4; byteShift+=4) {
207  const quint16 tmp = crcTable[checksum & 0xF];
208  checksum = (checksum >> 4) & 0x0FFF;
209  checksum = checksum ^ tmp ^ crcTable[(byte >> byteShift) & 0xF];
210  }
211  }
212  return checksum;
213 }

288 template<class T> bool FitStreamReaderPrivate::parseFileHeader()
289 {
290  Q_ASSERT(protocolVersion.isNull());
291  Q_ASSERT(profileVersion.isNull());
292 
293  // Read the header bytes.
294  const QByteArray header = readFileHeader<T>();
295  qDebug() << "Header size" << header.size() << "bytes";
296  if (header.isEmpty()) {
297  /// @todo set not-enough-bytes; ie might need to wait for more bytes.
298  qDebug() << "not enough bytes for header";
299  return false;
300  }
301  if (header.size() < 12) {
302  /// @todo set invalid header size error; ie FIT stream is corrupt / invalid.
303  qDebug() << "invalid header size";
304  return false;
305  }
306 
307  { // Protocol version is split into two parts: high 4 bits major, a low 4 bits minor.
308  const quint8 version = header.at(1);
309  protocolVersion = QVersionNumber(version >> 4, version & 0x0F);
310  qDebug() << "Protocol version" << protocolVersion;
311  }
312 
313  { // Profile version is major*100 + minor (ie minor could not be more than 99).
314  const quint16 version = qFromLittleEndian<quint16>(header.mid(2,2).data());
315  profileVersion = QVersionNumber(version/100, version%100);
316  qDebug() << "Profile version" << profileVersion;
317  }
318 
319  expectedDataSize = qFromLittleEndian<quint32>(header.mid(4,4).data());
320  qDebug() << "Data size" << expectedDataSize << "bytes";
321 
322  const QByteArray dataType = header.mid(8,4);
323  qDebug() << "Data type" << dataType;
324  if (dataType != QByteArray(".FIT")) {
325  /// @todo set invalid header data type (must be ".FIT").
326  return false;
327  }
328 
329  // Check the header's checksum (only present in 14+ byte headers, and even then may be 0x0000).
330  if (header.size() >= 14) {
331  const quint16 expectedChecksum = qFromLittleEndian<quint16>(data.mid(12,2).data());
332  if (expectedChecksum == 0x0000) {
333  qDebug() << "FIT file has (optional) checksum 0x0000; ignoring.";
334  } else {
335  const quint16 calculatedChecksum = fitChecksum(header.mid(0,12));
336  qDebug() << "Header checksum" << expectedChecksum << calculatedChecksum;
337  if (calculatedChecksum != expectedChecksum) {
338  qWarning() << "Checksum failure:" << calculatedChecksum << "!=" << expectedChecksum;
339  /// @todo set error, and return false here?
340  } else {
341  qDebug() << "Checkums match";
342  }
343  }
344  }
345  return true;
346 }

348 template<class T> bool FitStreamReaderPrivate::parseDefinitionMessage()
349 {
350  // Parse the record header.
351  qDebug() << "parsing definition message";
352  Q_ASSERT(bytesAvailable<T>());
353  const quint8 recordHeader = peekByte<T>();
354  qDebug() << __func__ << "recordHeader" << recordHeader;
355  Q_ASSERT_X(isDefinitionMessage(recordHeader), "parseDefinitionMessage",
356  "FIT record header does not indiciate a definition message");
357  const bool hasDevFields = (recordHeader & (1 << 5));
358  qDebug() << __func__ << "has dev fields" << hasDevFields;
359  qDebug() << __func__ << "reserved" << (recordHeader & (1 << 4));
360  const quint8 localMessageType = (recordHeader & 0xF); // Least significant 4 bits.
361  qDebug() << __func__ << "local message type" << localMessageType;
362 
363  // Read the completed definition record (if available).
364  if (bytesAvailable<T>() < 6) {
365  return false; // Not enough bytes.
366  }
367  const quint8 numberOfFields = peekByte<T>(5);
368  qDebug() << __func__ << "number of fields" << numberOfFields;
369  int numberOfDevFields = 0;
370  if (hasDevFields) {
371  const int offsetToNumberOfDevFields = 6 + (numberOfFields * 3);
372  qDebug() << __func__ << "offset to number of dev fields" << offsetToNumberOfDevFields;
373  if (bytesAvailable<T>() < offsetToNumberOfDevFields) {
374  return false; // Not enough bytes.
375  }
376  numberOfDevFields = peekByte<T>(offsetToNumberOfDevFields);
377  }
378  const int recordSize = 6 + (numberOfFields * 3) + (hasDevFields ? 1 + (numberOfDevFields * 3) : 0);
379  qDebug() << __func__ << "number of dev fields" << numberOfDevFields;
380  qDebug() << __func__ << "record size" << recordSize;
381  const QByteArray record = readBytes<T>(recordSize);
382  if (record.isEmpty()) {
383  return false; // Not enough bytes.
384  }
385 
386  // Parse the rest of the definition record.
387  DataDefinition defn;
388  defn.recordSize = 0;
389  defn.architecture = static_cast<Architecture>(record.at(2));
390  qDebug() << __func__ << "architecture" << (int)defn.architecture;
391  qDebug() << record.mid(3,2);
392  defn.globalMessageNumber = static_cast<MesgNum>((defn.architecture == Architecture::BigEndian)
393  ? qFromBigEndian<quint16>(record.mid(3,2))
394  : qFromLittleEndian<quint16>(record.mid(3,2)));
395  qDebug() << __func__ << "header" << (quint8)record.at(0);
396  qDebug() << __func__ << "reserved" << (quint8)record.at(1);
397  qDebug() << __func__ << "msgNum" << (int)defn.globalMessageNumber;
398 
399  // Parse the definition fields.
400  for (int i=0; i < numberOfFields; ++i) {
401  const int pos = 6 + (i * 3);
402  FieldDefinition field;
403  field.number = record.at(pos);
404  field.size = record.at(pos+1);
405  field.baseType = static_cast<FitBaseType>(record.at(pos+2));
406  qDebug() << __func__ << "field" << i << "number" << field.number;
407  qDebug() << __func__ << "field" << i << "size" << field.size;
408  qDebug() << __func__ << "field" << i << "type" << static_cast<quint8>(field.baseType);
409  defn.fieldDefinitions.append(field);
410  defn.recordSize += field.size;
411  }
412  Q_ASSERT(defn.fieldDefinitions.size() == numberOfFields);
413  qDebug() << __func__ << "defn record size" << defn.recordSize;
414 
415  // Parse the definition fields.
416  Q_ASSERT(hasDevFields || (numberOfDevFields == 0));
417  for (int i=0; i < numberOfDevFields; ++i) {
418  const int pos = 6 + (numberOfFields * 3) + 1 + (i * 3);
419  DeveloperFieldDefinition field;
420  field.fieldNumber = record.at(pos);
421  field.size = record.at(pos+1);
422  field.devDataIndex = record.at(pos+2);
423  qDebug() << __func__ << "dev field" << i << "number" << field.fieldNumber;
424  qDebug() << __func__ << "dev field" << i << "size" << field.size;
425  qDebug() << __func__ << "dev field" << i << "dataIndex" << field.devDataIndex;
426  defn.developerFieldDefinitions.append(field);
427  defn.recordSize += field.size;
428  }
429  Q_ASSERT(defn.developerFieldDefinitions.size() == numberOfDevFields);
430  qDebug() << __func__ << "defn record size" << defn.recordSize;
431  if (defn.recordSize == 0) {
432  qWarning() << "defintion record size is zero";
433  }
434 
435  // Record the definition data for future Data Messages.
436  dataDefinitions.insert(localMessageType, defn);
437  return true;
438 }

440 template<class T> FitDataMessage * FitStreamReaderPrivate::parseDataMessage()
441 {
442  Q_ASSERT(bytesAvailable<T>());
443  qDebug() << "parsing data message";
444  const quint8 recordHeader = peekByte<T>();
445  qDebug() << __func__ << "recordHeader" << recordHeader;
446  Q_ASSERT_X(!isDefinitionMessage(recordHeader), "parseDataMessage",
447  "FIT record header does not indiciate a data message");
448 
449  // Parse the record header.
450  quint8 localMessageType;
451  if (recordHeader & (1 << 7)) { // Compressed Timestamp Header.
452  qDebug() << __func__ << "Compressed Timestamp Header";
453  localMessageType = ((recordHeader >> 4) & 0x7);
454  const quint8 timeOffset = (recordHeader & 0xF);
455  qDebug() << "time offset" << timeOffset;
456  /// \todo Process timeOffset.
457  } else { // Normal (Data Message) Header
458  qDebug() << __func__ << "Normal (Data Message) Header";
459  qDebug() << __func__ << "reserved5" << (recordHeader & (1 << 5));
460  qDebug() << __func__ << "reserved4" << (recordHeader & (1 << 4));
461  localMessageType = (recordHeader & 0xF); // Least significant 4 bits.
462  }
463  qDebug() << __func__ << "local message type" << localMessageType;
464 
465  // Lookup the record's field definitions.
466  if (!dataDefinitions.contains(localMessageType)) {
467  qWarning() << "No definition for local message type" << localMessageType;
468  /// \todo Set error code.
469  return nullptr; // FIT data is corrupt; we cannot safely continue parsing.
470  }
471  const DataDefinition defn = dataDefinitions.value(localMessageType);
472 
473  // Parse record's Data Fields.
474  qDebug() << __func__ << "record size" << defn.recordSize;
475  if (defn.recordSize == 0) {
476  qWarning() << "record size is zero"; // Not really sure what to do here.
477  return nullptr;
478  }
479  const QByteArray record = readBytes<T>(defn.recordSize+1);
480  if (record.isEmpty()) {
481  return nullptr; // Not enough bytes.
482  }
483  qDebug() << "record" << record.mid(1);
484  return FitDataMessage::fromData(&defn, record.mid(1));
485 }

518 template<class T> FitDataMessage * FitStreamReaderPrivate::readNextDataMessage()
519 {
520  // If we haven't parsed the FIT File Header yet, do so now.
521  if ((protocolVersion.isNull() && (!parseFileHeader<T>()))) {
522  return nullptr;
523  }
524 
525  // Process all FIT Data Records until we get a FIT Data Message (or run out of bytes).
526  while (bytesAvailable<T>()) { // At least one byte, for the next data record header byte.
527  const quint8 recordHeader = peekByte<T>();
528  if (isDefinitionMessage(recordHeader)) {
529  if (!parseDefinitionMessage<T>()) return nullptr;
530  // Not returning here; we'll continue processing until we get a FIT Data Message.
531  } else return parseDataMessage<T>();
532  }
533  return nullptr;
534 }

536 bool FitStreamReaderPrivate::isDefinitionMessage(const quint8 recordHeader)
537 {
538  // For definition messages, bit 7 must be off (bit 7 on would indicate a Compressed Timestamp
539  // Data Message, which cannot be a Definition Message), and bit 6 must be on (otherwise this
540  // woud be a Normal Data Message, not a Definition Message).
541  return ((recordHeader >> 6) == 1); // Match bit 7 on and 6 off; ie 01xxxxxx.
542 }
