29 /*!
30  * Architecture Type for FIT Data Messages.
31  *
32  * Currently the FIT format only specified big and little endian byte-ordering, but the architecture
33  * is specified as a unsigned byte, so could, theoretically, support other architectures in future.
34  *
35  * \sa https://developer.garmin.com/fit/protocol/#definitionmessage
36  */
37 enum class Architecture : quint8 {
38  LittleEndian = 0, ///< Big-endian byte ordering.
39  BigEndian = 1, ///< Little-endian byte ordering.
40 };
41 
42 /*!
43  * Custom developer field definition.
44  *
45  * Typically, these definitions are extracted from FIT Definition Messages, then used to interpret
46  * subsequent FIT Data Messages.
47  *
48  * \sa https://developer.garmin.com/fit/protocol/#developerdatafielddescription
49  */
50 struct DeveloperFieldDefinition {
51  quint8 fieldNumber; ///< Maps to the field_definition_number of a field_description Message.
52  quint8 size; ///< Size (in bytes) of the specified FIT message’s field.
53  quint8 devDataIndex; ///< Maps to the developer_data_index of a developer_data_id Message.
54 };
55 
56 /*!
57  * Field Definition for FIT Data Messages.
58  *
59  * Typically, these definitions are extracted from FIT Definition Messages, then used to interpret
60  * subsequent FIT Data Messages.
61  *
62  * \sa https://developer.garmin.com/fit/protocol/#fielddefinition
63  */
64 struct FieldDefinition {
65  quint8 number; ///< Unique ID for the FIT field within a given FIT data message.
66  quint8 size; ///< Size (in bytes) of the field.
67  FitBaseType baseType; ///< Base type for interpreting unknown fields.
68 };
69 
70 /*!
71  * Data Message definition.
72  *
73  * Typically, these definitions are extracted from FIT Definition Messages, then used to interpret
74  * subsequent FIT Data Messages.
75  *
76  * \sa https://developer.garmin.com/fit/protocol/#fielddefinition
77  */
78 struct DataDefinition {
79  Architecture architecture; ///< Architecture type for any multi-byte fields.
80  MesgNum globalMessageNumber; ///< FIT Global Message Number the Data Message represents.
81 
82  /// Definitons list of all fields, if any, present in the described Data Message.
83  QList<FieldDefinition> fieldDefinitions;
84 
85  /// Definitions list of all custom fields, if any, present in the described Data Message.
86  QList<DeveloperFieldDefinition> developerFieldDefinitions;
87 
88  /*!
89  * Total size, in byetes, of all fields in the described Data Message.
90  *
91  * This is the combined sum of all sizes in the \c fieldDefinitions and
92  * \c developerFieldDefintions members, which is calculated and stored when then definition is
93  * initially populated (by FitStreamReaderPrivate::parseDefinitionMessage) as an optimisation
94  * for code reading the definitions.
95  */
96  int recordSize;
97 };
