101 quint16 fitChecksum(const QByteArray &data) {
102  quint16 checksum=0;
103  for (const auto &byte: data) {
104  static const quint16 crcTable[16] = {
105  0x0000, 0xCC01, 0xD801, 0x1400, 0xF001, 0x3C00, 0x2800, 0xE401,
106  0xA001, 0x6C00, 0x7800, 0xB401, 0x5000, 0x9C01, 0x8801, 0x4400
107  };
108  for (int byteShift=0; byteShift<=4; byteShift+=4) {
109  const quint16 tmp = crcTable[checksum & 0xF];
110  checksum = (checksum >> 4) & 0x0FFF;
111  checksum = checksum ^ tmp ^ crcTable[(byte >> byteShift) & 0xF];
112  }
113  }
114  return checksum;
115 }

159 template<class T> bool FitStreamReaderPrivate::parseFileHeader()
160 {
161  Q_ASSERT(protocolVersion.isNull());
162  Q_ASSERT(profileVersion.isNull());
163 
164  // Read the header bytes.
165  const QByteArray header = readFileHeader<T>();
166  qDebug() << "Header size" << header.size() << "bytes";
167  if (header.isEmpty()) {
168  /// @todo set not-enough-bytes; ie might need to wait for more bytes.
169  qDebug() << "not enough bytes for header";
170  return false;
171  }
172  if (header.size() < 12) {
173  /// @todo set invalid header size error; ie FIT stream is corrupt / invalid.
174  qDebug() << "invalid header size";
175  return false;
176  }
177 
178  { // Protocol version is split into two parts: high 4 bits major, a low 4 bits minor.
179  const quint8 version = header.at(1);
180  protocolVersion = QVersionNumber(version >> 4, version & 0x0F);
181  qDebug() << "Protocol version" << protocolVersion;
182  }
183 
184  { // Profile version is major*100 + minor (ie minor could not be more than 99).
185  const quint16 version = qFromLittleEndian<quint16>(header.mid(2,2).data());
186  profileVersion = QVersionNumber(version/100, version%100);
187  qDebug() << "Profile version" << profileVersion;
188  }
189 
190  expectedDataSize = qFromLittleEndian<quint32>(header.mid(4,4).data());
191  qDebug() << "Data size" << expectedDataSize << "bytes";
192 
193  const QByteArray dataType = header.mid(8,4);
194  qDebug() << "Data type" << dataType;
195  if (dataType != QByteArray(".FIT")) {
196  /// @todo set invalid header data type (must be ".FIT").
197  return false;
198  }
199 
200  // Check the header's checksum (only present in 14+ byte headers, and even then may be 0x0000).
201  if (header.size() >= 14) {
202  const quint16 expectedChecksum = qFromLittleEndian<quint16>(data.mid(12,2).data());
203  if (expectedChecksum == 0x0000) {
204  qDebug() << "FIT file has (optional) checksum 0x0000; ignoring.";
205  } else {
206  const quint16 calculatedChecksum = fitChecksum(header.mid(0,12));
207  qDebug() << "Header checksum" << expectedChecksum << calculatedChecksum;
208  if (calculatedChecksum != expectedChecksum) {
209  qWarning() << "Checksum failure:" << calculatedChecksum << "!=" << expectedChecksum;
210  /// @todo set error, and return false here?
211  } else {
212  qDebug() << "Checkums match";
213  }
214  }
215  }
216  return true;
217 }

257 {
258  // Return `n` bytes, where `n` is given by the first byte (ie a length-prefixed buffer).
259  return (bytesAvailable<T>()) ? readBytes<T>(peekByte<T>()) : QByteArray();

262 template<class T> FitDataMessage FitStreamReaderPrivate::readNextDataMessage()
263 {
264  // If we haven't parsed the FIT File Header yet, do so now.
265  if ((protocolVersion.isNull() && (!parseFileHeader<T>()))) {
266  return FitDataMessage(); // Need a way to return a null message.
267  }
268 
269  // Process all FIT Data Records until we get a FIT Data Message (or run out of bytes).
270  while (bytesAvailable<T>()) { // At least one byte, for the next data record header byte.
271  const quint8 recordHeader = peekByte<T>();
272  if (recordHeader & (1 << 6)) { // Bit 6 indicates a Definition Message.
273  if (!parseDefinitionMessage<T>()) return FitDataMessage();
274  // Not returning here; we'll continue processing until we get a FIT Data Message.
275  } else return parseDataMessage<T>();
276  }
277  return FitDataMessage();
278 }
